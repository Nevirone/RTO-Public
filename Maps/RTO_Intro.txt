component ActivateIDZone
{
    Description ="Activate other objects when a player enters this zone";

    Ids = "";
    IdsTooltip = "IDs that will get activated. (separated by whitespace)";

    DeactivateIds = "";
    DeactivateIdsTooltip = "IDs that will get deactivated. (separated by whitespace)";

    PlayerType = "Human";
    PlayerTypeDropbox = "Human, Titan, Shifter";

    _activated = false;

    _activateOjectList = List();
    _deactivateObjectList = List();

    function Init()
    {
        if(String.Trim(self.Ids) != "")
        {
            for(ID in String.Split(self.Ids, " ")){
                obj = Map.FindMapObjectByID(Convert.ToInt(ID));
                if(obj != null){
                    self._activateOjectList.Add(obj);
                }
            }
        }
        if(String.Trim(self.DeactivateIds) != "")
        {
            for(ID in String.Split(self.DeactivateIds, " ")){
                obj = Map.FindMapObjectByID(Convert.ToInt(ID));
                if(obj != null){
                    self._deactivateObjectList.Add(obj);
                }
            }
        }
    }

    function OnCollisionEnter(obj)
    {
        if(self._activated){return;}

        if(self.PlayerType == "Human"){
            if(Game.PlayerHumans.Contains(obj)){
                self._activated = true;
            }
        }
        elif(self.PlayerType == "Titan"){
            if(Game.PlayerTitans.Contains(obj)){
                self._activated = true;
            }
        }
        elif(self.PlayerType == "Shifter"){
            if(Game.PlayerShifters.Contains(obj)){
                self._activated = true;
            }
        }
        if(self._activated){
            for(mapObject in self._activateOjectList){
                mapObject.Active = true;
            }
            for(mapObject in self._deactivateObjectList){
                mapObject.Active = false;
            }
        }
    }
}
